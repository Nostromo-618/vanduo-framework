Usage Example

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>My Website</title>
    <link rel="stylesheet" href="css/vanduo.css">
</head>
<body>
    <div class="container">
        <div class="row">
            <div class="col-12 col-md-6">
                <h1 class="text-primary">Hello World</h1>
                <p class="text-secondary">This is a paragraph.</p>
            </div>
            <div class="col-12 col-md-6">
                <div class="bg-primary p-4 text-white rounded">
                    Content here
                </div>
            </div>
        </div>
    </div>
    <script src="js/vanduo.js"></script>
</body>
</html>
